script_template: |
  #!/usr/bin/env node

  /**
   * {SCRIPT_NAME} — {DESCRIPTION}
   *
   * Usage:
   *   node {filename}.js
   *   node {filename}.js --headless
   *   node {filename}.js --model "openai/gpt-4o"
   *
   * Generated by: Playwright Architect Squad
   * Date: {DATE}
   */

  import { Stagehand } from '@browserbasehq/stagehand';
  import { z } from 'zod';
  import { readFileSync, writeFileSync, existsSync } from 'node:fs';
  import { resolve, dirname } from 'node:path';
  import { fileURLToPath } from 'node:url';

  const __dirname = dirname(fileURLToPath(import.meta.url));

  // ── Config ──────────────────────────────────────────────────────────────
  function loadEnv() {
    const envPath = resolve(__dirname, '../../tools/smart-browser-playwright/.env');
    if (!existsSync(envPath)) return;
    const lines = readFileSync(envPath, 'utf-8').split('\n');
    for (const line of lines) {
      const trimmed = line.trim();
      if (!trimmed || trimmed.startsWith('#')) continue;
      const eqIndex = trimmed.indexOf('=');
      if (eqIndex === -1) continue;
      const key = trimmed.slice(0, eqIndex).trim();
      const val = trimmed.slice(eqIndex + 1).trim();
      if (!process.env[key]) process.env[key] = val;
    }
  }

  loadEnv();

  const HEADED = !process.argv.includes('--headless');
  const MODEL = process.argv.find((_, i, a) => a[i-1] === '--model') || 'openai/gpt-4o-mini';

  // ── Schemas ─────────────────────────────────────────────────────────────
  {ZOD_SCHEMAS}

  // ── Main ────────────────────────────────────────────────────────────────
  async function main() {
    const apiKey = process.env.OPENAI_API_KEY;
    if (!apiKey) {
      console.error('OPENAI_API_KEY not set. Check tools/smart-browser-playwright/.env');
      process.exit(1);
    }

    const stagehand = new Stagehand({
      env: 'LOCAL',
      localBrowserLaunchOptions: {
        headless: !HEADED,
        viewport: { width: 1280, height: 720 },
      },
      apiKey,
      model: { modelName: MODEL, apiKey },
    });

    try {
      await stagehand.init();
      const page = stagehand.page;
      console.log('Browser initialized');

      {SCRIPT_BODY}

    } catch (err) {
      console.error(`Error: ${err.message}`);
      process.exit(1);
    } finally {
      await stagehand.close();
      console.log('Done.');
    }
  }

  main();

implementation_patterns:
  navigate:
    code: |
      await page.goto('{url}', { waitUntil: 'domcontentloaded' });

  act_single:
    code: |
      await stagehand.act({ action: '{description}' });

  extract_data:
    code: |
      const result = await stagehand.extract({
        instruction: '{instruction}',
        schema: {SchemaName},
      });

  agent_task:
    code: |
      const agent = stagehand.agent({ modelName: MODEL });
      const result = await agent.execute('{task_description}');

  login_flow:
    code: |
      await page.goto('{login_url}', { waitUntil: 'domcontentloaded' });
      await stagehand.act({ action: 'type "{email}" in the email field' });
      await stagehand.act({ action: 'type "{password}" in the password field' });
      await stagehand.act({ action: 'click the login or sign in button' });
      await page.waitForNavigation({ waitUntil: 'domcontentloaded', timeout: 30000 });

  scroll_collect:
    code: |
      let allItems = [];
      let prevHeight = 0;
      for (let i = 0; i < {max_scrolls}; i++) {
        const extracted = await stagehand.extract({
          instruction: '{instruction}',
          schema: {SchemaName},
        });
        allItems.push(...extracted.items);
        const newHeight = await page.evaluate(() => document.body.scrollHeight);
        if (newHeight === prevHeight) break;
        prevHeight = newHeight;
        await page.evaluate(() => window.scrollTo(0, document.body.scrollHeight));
        await page.waitForTimeout(1500);
      }

  download_file:
    code: |
      const [download] = await Promise.all([
        page.waitForEvent('download'),
        stagehand.act({ action: '{click_description}' }),
      ]);
      const filePath = resolve(__dirname, '{output_path}', download.suggestedFilename());
      await download.saveAs(filePath);

  save_output:
    code: |
      const outputPath = resolve(__dirname, '{output_file}');
      writeFileSync(outputPath, JSON.stringify(result, null, 2));
      console.log('Saved to:', outputPath);
